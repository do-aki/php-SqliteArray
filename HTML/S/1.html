 546     private function _intersectKey_array($rhs) {

 614         }
 615         $sth->closeCursor();

 618         $this->last_find_value = $result;

 679      * オブジェクト生成の準備を行う
 680      *
 681      * !コンストラクタで必ず呼ばなければならない!
 682      *  テーブル名の設定と、プログラムの最後にリソースを回収するための仕掛け(ルートオブジェクトの保持)を施している
 683      *
 684      * @param SqliteArray $self
 685      * @return void
 686      */
 687     private static function _prepareObject($self) {
 688         if (!isset(self::$root_object)) {
 689             self::$root_object = false;
 690             new self(); // ルートオブジェクトの生成 (コンストラクタを介して (*1) へ)
 691         } elseif (!self::$root_object) {
 692             // ルートオブジェクトの登録 (*1)
 693             self::$root_object = $self;
 694             return;
 695         }
 696 
 697         // 通常オブジェクトとしての登録
 698         $self->table_name = self::TABLE_PREFIX . ++self::$object_counter;
 699     }

 701     /**
 702      * 配列の添え字として有効なキーを返す
 703      *
 704      * @param $key
 705      * @return integer or string
 706      */
 707     private static function _convertKey($key) {
 708         if (self::_isNumericKey($key)) {
 709             $key = intval($key);
 710         }
 711         return $key;
 712     }

 714     /**
 715      * 数添え字として見なせるか判別
 716      *
 717      * @param mixed $key 添え字
 718      * @return boolean
 719      */
 720     private static function _isNumericKey($key) {
 721         return is_int($key) || is_bool($key) || is_float($key) || preg_match('/\A(:?0|[1-9][0-9]*)\Z/', $key);
 722     }
